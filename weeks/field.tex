\chapter{Fields}

\textbf{Recall:} $\mathbb{F}$ is a field if $\mathbb{F}$ is an integral domain, and for all $x \in \mathbb{F} \setminus \{0\}$, there exists a multiplicative inverse $x^{-1} \in \mathbb{F} \setminus \{0\}$. In general, there are 3 kinds of fields:
\begin{enumerate}
    \item \textbf{Number field} (a field `in between' $\mathbb{Q}$ and $\mathbb{C}$) -- for example, $\mathbb{Q}$, $\mathbb{R}$, $\mathbb{Q}(i)$, $\dots$
    \item \textbf{Finite field} (a field $\mathbb{F}$ with $|\mathbb{F}| < \infty$) -- for example, $\mathbb{Z}_p = \mathbb{F}_p$ for $p$ prime. $\mathbb{F}_2[x]/\langle x^3+x+1 \rangle$ has $8=2^3$ elements.
    \item \textbf{Function fields} -- for example, 
    $$\mathbb{C}(x) = \text{Frac}(\mathbb{C}[x]) = \{ f(x)/g(x) \mid f(x), g(x) \in \mathbb{C}[x], g(x) \neq 0 \}$$
\end{enumerate}
We will only focus on (1) or (2).

\section{Field Extension and Degree}

\textbf{Motivation:} Given a field $\mathbb{F}$, construct a ``bigger'' field $\mathbb{K}$ that "contains" $\mathbb{F}$. For example, by the discussions at the polynomial ring section, let $\mathbb{F} = \mathbb{F}_2$ and $\mathbb{K} = \mathbb{F}_2[x]/\langle x^3+x+1 \rangle$.

Since $x^3+x+1$ is irreducible in $\mathbb{F}_2[x]$, we have an injective unital ring (or field) homomorphism $i: \mathbb{F} \to \mathbb{K}$ where $i(0) = 0 + \langle x^3+x+1 \rangle$ and $i(1) = 1 + \langle x^3+x+1 \rangle$. 

\begin{definition}
Let $\mathbb{F}$ and $\mathbb{K}$ be fields. We say $\mathbb{K}$ is a \textbf{field extension} of $\mathbb{F}$ if there is an injective unital field homomorphism $i: \mathbb{F} \hookrightarrow  \mathbb{K}$.
\end{definition}

In this case, $\mathbb{K}$ is a vector space over $\mathbb{F}$. More explicitly, there is a scalar multiplication map and vector addition $$\cdot: \mathbb{F} \times \mathbb{K} \to \mathbb{K} 
\quad \quad +: \mathbb{K} \times \mathbb{K} \to \mathbb{K}$$
defined by $\alpha \cdot x = i(\alpha)x$ for $\alpha \in \mathbb{F}, x \in \mathbb{K}$, and $+$ is the usual addition in $\mathbb{K}$. 

From now on, we write
\begin{center}
    $\mathbb{F} \subseteq \mathbb{K}$ (by abuse of notations) or $\mathbb{K} : \mathbb{F}$
\end{center}
for field extensions.

\begin{definition}
Let $\mathbb{K} : \mathbb{F}$ be a field extension. Its \textbf{degree} of extension is defined as:
\[ [\mathbb{K}:\mathbb{F}] := \dim_{\mathbb{F}}(\mathbb{K})\]    
as a vector space over $\mathbb{F}$.
\end{definition}


\begin{example}
\begin{enumerate}
    \item $\mathbb{F}=\mathbb{R}$, $\mathbb{K}=\mathbb{C}$, with the inclusion $i: \mathbb{R} \hookrightarrow \mathbb{C}$. Then $[\mathbb{C}:\mathbb{R}] = 2$, since $$\mathbb{C} = \{a+bi \mid a,b \in \mathbb{R}\} = \text{Span}_{\mathbb{R}}\{1, i\}.$$
    Thus, $\dim_{\mathbb{R}}(\mathbb{C}) = 2$ (since $\{1,i\}$ is linearly independent over $\mathbb{R}$).

    \item $\mathbb{F}=\mathbb{Q}$, $\mathbb{K}=\mathbb{R}$, with the inclusion $i: \mathbb{Q} \hookrightarrow \mathbb{R}$. \\
    To find $[\mathbb{R}:\mathbb{Q}]$, consider that $\mathbb{R} = \text{Span}_{\mathbb{Q}}\{1, \sqrt{2}, \sqrt{3}, e, \pi, \dots\}$. \\
    The set of transcendental numbers like $e, \pi$ is infinite and they are linearly independent over $\mathbb{Q}$.
    Thus, $[\mathbb{R}:\mathbb{Q}] = \infty$.
\end{enumerate}
\end{example}

Here is an important example of field extension:
\begin{theorem}[Kronecker] \label{thm:kronecker}
Let $\mathbb{F}$ be a field, and $p(x) \in \mathbb{F}[x]$ be an irreducible polynomial of degree $m$.
Then $\mathbb{K} = \mathbb{F}[x]/\langle p(x) \rangle$ is a field extension of $\mathbb{F}$ such that 
$$[\mathbb{K}:\mathbb{F}]=m.$$
Moreover, there exists an element $\alpha \in \mathbb{K}$ such that:
\begin{itemize}
    \item $\mathbb{K} = \mathrm{Span}_{\mathbb{F}}\{1, \alpha, \dots, \alpha^{m-1}\}$, i.e. $\{1, \alpha, \dots, \alpha^{m-1}\}$ is a basis of $\mathbb{K}$ as a $\mathbb{F}$-vector space; and
    \item $p(\alpha) = 0$ in $\mathbb{K}$.
\end{itemize}
\end{theorem}

\begin{proof}
Let $p(x) = b_mx^m + \dots + b_1 x + b_0 \in \mathbb{F}[x]$. Then the ideal $\langle p(x) \rangle$ is a maximal in $\mathbb{F}[x]$.
This implies that $\mathbb{K} = \mathbb{F}[x]/\langle p(x) \rangle$ is a field, and
$$i: \mathbb{F} \hookrightarrow \mathbb{K}, \quad \quad i(a) := a + \langle p(x) \rangle.$$
is a field extension.

Now we study the dimension $\dim_{\mathbb{F}}(\mathbb{K})$. Consider the element $\alpha := x + \langle p(x) \rangle \in \mathbb{K}$. Then for 
\begin{center}
$p(x) = b_m x^m + \dots  + b_0 \in \mathbb{K}[x]$\quad (more precisely, $p(x) = i(b_m) x^m + \dots + i(b_0) \in \mathbb{K}[x]$),
\end{center}
one has
\begin{align*}
p(\alpha) &= b_m \alpha^m + \dots + b_1 \alpha + b_0 \\
&= b_m(x+\langle p(x) \rangle)^m + \dots + b_1(x+\langle p(x) \rangle) + b_0(1+\langle p(x) \rangle) \\
&= (b_m x^m + \dots + b_1 x + b_0) + \langle p(x) \rangle \\
&= p(x) + \langle p(x) \rangle = 0 + \langle p(x) \rangle = 0_K
\end{align*}
So, the second part of the theorem is proved. Moreover, this also implies that 
\[\alpha^m = -\frac{1}{b_m}(b_{m-1}\alpha^{m-1} + \dots + b_0)\] 
i.e. the set $\{1, \alpha, \dots, \alpha^m\}$ is linearly dependent in $\mathbb{K}$. More precisely, $\alpha^m$ is a linear combination of $\{1, \alpha, \dots, \alpha^{m-1}\}$.
Similarly, for any $n \ge m$, $\alpha^n$ is also a linear combination of $\{1, \alpha, \dots, \alpha^{m-1}\}$.

\textbf{Claim:} The set $\{1, \alpha, \dots, \alpha^{m-1}\}$ is a basis of $\mathbb{K}$ over $\mathbb{F}$.

\textbf{Spanning set:} Take any element $k \in \mathbb{K}$. By definition, $k = (c_l x^l + \dots + c_1 x + c_0) + \langle p(x) \rangle$ for some $c_i \in \mathbb{F}$.
This is equal to $c_l \alpha^l + \dots + c_1 \alpha + c_0$.
Since all powers $\alpha^j$ for $j \ge m$ can be reduced to a linear combination of $\{1, \alpha, \dots, \alpha^{m-1}\}$, any element $k \in \mathbb{K}$ can be written as a linear combination of these basis elements.
Thus, $\mathbb{K} = \text{Span}_{\mathbb{F}}\{1, \alpha, \dots, \alpha^{m-1}\}$.

\textbf{Linearly independent:} Suppose $d_{m-1}\alpha^{m-1} + \dots + d_1\alpha + d_0 = 0_{\mathbb{K}}$ for some $d_i \in \mathbb{F}$, not all zero.
This is equivalent to $(d_{m-1}x^{m-1} + \dots + d_1x + d_0) + \langle p(x) \rangle = 0_{\mathbb{K}}$.
Let $g(x) = d_{m-1}x^{m-1} + \dots + d_0$.
The equation means $g(x) \in \langle p(x) \rangle$.
This implies that $p(x)$ divides $g(x)$, i.e., $g(x) = p(x) \cdot r(x)$ for some $r(x) \in \mathbb{F}[x]$.
But this is a contradiction, because $\deg(g(x)) \le m-1$ while $\deg(p(x)) = m$. The degree of a non-zero polynomial $g(x)$ cannot be less than the degree of a non-zero polynomial $p(x)$ that divides it. Thus, $g(x)$ must be the zero polynomial, which means all $d_i$ must be zero.
The set is linearly independent.

Since $\{1, \alpha, \dots, \alpha^{m-1}\}$ is a basis for $\mathbb{K}$ over $\mathbb{F}$, the dimension is $m$.
Therefore, $[\mathbb{K}:\mathbb{F}] = m$.
\end{proof}

\begin{example}
Let $\mathbb{F}=\mathbb{Q}$, $p(x)=x^3-2 \in \mathbb{Q}[x]$ (irreducible by Eisenstein's criterion and Gauss' Lemma).
Then $[\mathbb{K}:\mathbb{Q}]=3$ for $\mathbb{K}=\mathbb{Q}[x]/\langle x^3-2 \rangle$.
Let $\alpha = x + \langle x^3-2 \rangle$. Then $$\mathbb{K} = \text{Span}_{\mathbb{Q}}\{1, \alpha, \alpha^2\} = \{a_2\alpha^2+a_1\alpha+a_0 \mid a_i \in \mathbb{Q}\}.$$
We know $\alpha^3-2=0$, so $\alpha^3=2$, i.e.
$\mathbb{K}$ is a field that `contains' $\mathbb{Q}$ and a cube root of 2. We also know how to do arithmetic in $\mathbb{K}$, for instance:
\begin{align*}
(2\alpha^2+\tfrac{3}{4}) \cdot (\alpha-6) &= 2\alpha^3 - 12\alpha^2 + \tfrac{3}{4}\alpha - \tfrac{18}{4} \\
&= 2(2) - 12\alpha^2 + \tfrac{3}{4}\alpha - \tfrac{9}{2} \\
&= 4 - 12\alpha^2 + \tfrac{3}{4}\alpha - \tfrac{9}{2} \\
&= -12\alpha^2 + \tfrac{3}{4}\alpha - \tfrac{1}{2}
\end{align*}
\textbf{Exercise:} What is $\alpha^{-1} \in \mathbb{K}$?
\end{example}

In the above example, By taking $\mathbb{K} = \mathbb{Q}[x]/\langle x^3-2\rangle : \mathbb{Q}$, Kronecker's theorem implies that $x^3 -2$ has a root $\alpha \in \mathbb{K}$. So Factor Theorem implies:
$$x^3 -2 = (x-\alpha)(x^2 + \alpha x + \alpha^2) \in \mathbb{K}[x]$$ 
However, $\mathbb{K}$ does not contain all the roots of $p(x)$. So we have to construct a larger field $\mathbb{E} : \mathbb{K}$. To do so, consider $\mathbb{E} = \mathbb{K}[x]/\langle x^2+\alpha x+\alpha^2 \rangle : \mathbb{K}$. Then $\mathbb{E}$ must contain a root $\beta$ of the polynomial $x^2+\alpha x+\alpha^2 \in \mathbb{K}[x] \subseteq \mathbb{E}[x]$, i.e.
$$(x-\beta) \mid x^2+\alpha x+\alpha^2 \quad \Longrightarrow \quad x^2+\alpha x+\alpha^2 = (x-\beta)(x-\gamma) \in \mathbb{E}[x].$$
Therefore,
$$x^3-2 = (x-\alpha)(x-\beta)(x-\gamma) \in \mathbb{E}[x],$$
i.e. $\alpha, \beta, \gamma \in \mathbb{E}$ contains all the roots of $x^3 - 2$.

\begin{definition}
Let $\mathbb{F}$ be a field, and $p(x) \in \mathbb{F}[x]$. We say a field extension $\mathbb{E}:\mathbb{F}$ {\bf splits} $p(x)$ if it can be factorized linearly in $\mathbb{E}[x]$:
$$p(x) = a(x-a_1)\cdots(x-a_n)\ \ \mathrm{in}\ \ \mathbb{E}[x].$$
In other words, $\mathbb{E}$ contains all the roots of $p(x)$.
\end{definition}

In the above example above, $\mathbb{K}: \mathbb{Q}$ does not split $x^3-2$, while $\mathbb{E} : \mathbb{Q}$ splits $x^3 -2$. Also, all polynomials $f(x) \in \mathbb{R}[x]$ splits in $\mathbb{C}$. This is the celebrated Fundamental Theorem of Algebra!



\begin{proposition}
For any field $\mathbb{F}$ and any polynomial $p(x) \in \mathbb{F}[x]$, there exists a field extension $\mathbb{E}:\mathbb{F}$ such that $\mathbb{E}$ splits $p(x)$.
\end{proposition}
\begin{proof}
Prove by induction on the degree of $p(x)$ (for any base field $\mathbb{F}$). If $p(x) \in \mathbb{F}$ is of degree $1$, we are done. By induction hypothesis, assume the theorem holds for all polynomials of degree $< k$ over any field. Consider a polynomial $p(x)$ of degree $k$. Let $f(x)$ be an irreducible factor of $p(x)$.
Then take $\mathbb{K} = \mathbb{F}[x]/\langle f(x) \rangle$ so that $a := x + \langle f(x) \rangle \in \mathbb{K}$ is a root of $f(x)$ and thus of $p(x)$.
So $p(x) = (x-a) g(x)$ in $\mathbb{K}[x]$, where $\deg(g(x)) = k-1$.
By the induction hypothesis, there exists an extension $\mathbb{E}:\mathbb{K}$ such that $g(x)$ splits in $\mathbb{E}[x]$, say $g(x) = c(x-a_2)...(x-a_k)$. So we have $p(x) = c(x-a)(x-a_2)...(x-a_k)$ in $\mathbb{E}[x]$, and we are done.
\end{proof}

\section{Simple Extension}
In the previous section, we begin with a irreducible polynomial $p(x) \in \mathbb{F}[x]$,  and {\bf construct} a field extension $\mathbb{E}$ so that it contains some (or all) roots of $p(x)$.

In this section, we change our perspective a little bit – we {\bf fix} a field extension $\mathbb{L}:\mathbb{F}$, and see which elements
in $\mathbb{L}$ are roots of a polynomial $p(x) \in \mathbb{F}[x]$.

\begin{definition}
Let $\mathbb{F}$ be a field and $\mathbb{L}$ be a field extension of $\mathbb{F}$.
An element $\alpha \in \mathbb{L}$ is
\begin{itemize}
    \item \textbf{algebraic} if there exists a non-zero polynomial $p(x) \in \mathbb{F}[x]$ such that $p(\alpha)=0$ (in $\mathbb{L}$).
    \item Otherwise, $\alpha$ is called \textbf{transcendental}.
\end{itemize}
\end{definition}

\begin{example}
Let $\mathbb{F}=\mathbb{Q} \subseteq \mathbb{L}=\mathbb{C}$:
\begin{itemize}
    \item $i = \sqrt{-1}$ is algebraic over $\mathbb{Q}$ (its polynomial is $p(x) = x^2+1$).
    \item $\sqrt[4]{2}, \sqrt{3}$ are algebraic over $\mathbb{Q}$ (their polynomials are $p(x)=x^4-2$ and $x^2-3$, respectively).
    \item $\sqrt{2}+\sqrt{3}$ is algebraic over $\mathbb{Q}$ (Homework).
    \item $e, \pi$ are transcendental over $\mathbb{Q}$ (there is NO polynomial with $\mathbb{Q}$-coefficients having $e$ or $\pi$ as a root).
\end{itemize}
\end{example}

\begin{theorem}
If $\alpha$ is transcendental over $\mathbb{F}$, then $\mathbb{F}[\alpha] \cong \mathbb{F}[x]$ as rings (but not as fields).
\end{theorem}
\begin{proof}
    Consider the ring homomorphism $\phi: \mathbb{F}[x] \to \mathbb{F}[\alpha]$ given by
    $$\phi(p(x)) := p(\alpha)$$
    Since $\alpha$ is transcendental, $p(\alpha) \neq 0$ for any polynomial $p(x)$. Hence $\phi$ is an injective map. Also, $\phi$ is surjective by definition. Consequently, it is a ring isomorphism.
\end{proof}

As for $\beta$ algebraic:
\begin{proposition} \label{prop:minpoly}
Let $\beta \in \mathbb{L}$ be algebraic over $\mathbb{F}$. Consider the set $I = \{ g(x) \in \mathbb{F}[x] \mid g(\beta) = 0 \}$.
Then:
\begin{enumerate}
    \item $I$ is a non-zero ideal in $\mathbb{F}[x]$.
    \item $I = \langle p(x) \rangle$ for some unique monic polynomial $p(x) \in \mathbb{F}[x]$. (This is because $\mathbb{F}[x]$ is a PID).
    \item $p(x)$ is the unique monic polynomial of smallest degree in $I$.
    \item $p(x)$ is irreducible.
\end{enumerate}
This polynomial $m_{\beta}(x) := p(x)$ is called the \textbf{minimal polynomial} of $\beta$ over $\mathbb{F}$.
\end{proposition}

\begin{proof}
(a) Let's show $p(x)$ has the smallest degree. By the division algorithm, suppose on the contrary there exists $g(x) \in I$ with $\deg(g) < \deg(p)$. Since $I=\langle p(x) \rangle$, $g(x)$ must be a multiple of $p(x)$, which is impossible unless $g(x)=0$. Contradiction.

(b) Let's show $p(x)$ is irreducible. Suppose on the contrary that $p(x) = h_1(x)h_2(x)$ with $\deg(h_1) > 0$ and $\deg(h_2) > 0$.
Then $p(\beta) = h_1(\beta)h_2(\beta) = 0$ in the field $\mathbb{L}$.
This implies that either $h_1(\beta)=0$ or $h_2(\beta)=0$.
This means either $h_1(x) \in I$ or $h_2(x) \in I$.
But $\deg(h_1) < \deg(p)$ and $\deg(h_2) < \deg(p)$, which contradicts the fact that $p(x)$ is a polynomial of smallest degree in $I$. Thus, $p(x)$ must be irreducible.
\end{proof}


\begin{example}
Let $\mathbb{F} = \mathbb{Q}$ be the base field, and $\mathbb{L} = \mathbb{C}$.
\begin{itemize}
    \item For $\beta = i$, the minimal polynomial is $m_{\beta}(x) = x^2+1$.
    \item For $\beta = \sqrt[4]{2}$, the minimal polynomial is $m_{\beta}(x) = x^4-2$.
    \item For $\beta = e^{2\pi i/5}$, the minimal polynomial is $m_{\beta}(x) = x^4+x^3+x^2+x+1$. \\(Since $\beta^5=1$, and $x^5-1 = (x-1)(x^4+x^3+x^2+x+1)$).
\end{itemize}
\end{example}

\begin{theorem}
Let $\mathbb{L}:\mathbb{F}$ be a field extension, and $\beta \in \mathbb{L}$ be algebraic. Then the polynomial ring $\mathbb{F}[\beta]$ satisfies
$$\mathbb{F}[\beta] \cong \mathbb{F}[x]/\langle m_\beta(x) \rangle$$ 
and consequently it is a field.
\end{theorem}
\begin{proof}
Consider the surjective ring homomorphism $\phi: \mathbb{F}[x] \to \mathbb{F}[\beta]$ defined by evaluation at $\beta$:
\[ \phi(a_n x^n + \dots + a_0) = a_n \beta^n + \dots + a_0 \]
The kernel of this map is precisely the ideal $I = \{ f(x) \in \mathbb{F}[x] \mid \mathbb{F}(\beta)=0 \}$, which we know is equal to $\langle m_\beta(x) \rangle$.
So the result follows by the First Isomorphism Theorem for rings. 

Since $m_\beta(x)$ is irreducible, $\langle m_\beta(x) \rangle$ is a maximal ideal and hence $\mathbb{F}[x]/\langle m_\beta(x) \rangle \cong \mathbb{F}[\beta]$ is a field.
\end{proof}

Combining the above result with Kronecker's Theorem, one has
$$\mathbb{F}(\beta) := \mathbb{F}[\beta] = \mathrm{Span}_{\mathbb{F}}\{1,\beta,\dots, \beta^{\deg(m_{\beta}(x))-1}\}.$$
(we use round bracket $\mathbb{F}(\beta)$ instead of square bracket $\mathbb{F}[\beta] $ to emphasize that it is a field).

\begin{definition}
Under the above setting, the field extension 
$$\mathbb{F}(\beta): \mathbb{F}$$ 
is the \textbf{simple extension} of $\mathbb{F}$ by $\beta$. It is the smallest field extension of $\mathbb{F}$ containing $\beta$.
\end{definition}

We can also extend `non-simply' to any (finite) set of elements:
\begin{definition}
Let $\mathbb{L}:\mathbb{F}$ be a field extension, and $\alpha_1, \dots, \alpha_k \in \mathbb{L}$. Write 
$$\mathbb{F}(\alpha_1, \dots, \alpha_k): \mathbb{F}$$ 
as the smallest field extension containing $\alpha_1, \dots, \alpha_k$. In other words,
\[ \mathbb{F}(\alpha_1, \dots, \alpha_k) := \bigcap_{\mathbb{F} \subseteq \mathbb{K} \subseteq \mathbb{L}, \ \alpha_1, \dots, \alpha_k \in \mathbb{K}} \mathbb{K} \]
\end{definition}

\begin{proposition}
$\mathbb{F}(\alpha_1, \alpha_2, \alpha_3, \dots, \alpha_k) = (\mathbb{F}(\alpha_1))(\alpha_2, \alpha_3,\dots, \alpha_k) = (\mathbb{F}(\alpha_1,\alpha_2))(\alpha_3 \dots, \alpha_k) = \dots $.
\end{proposition}
\begin{proof}
Check the case of $k = 2$, i.e. $\mathbb{F}(\alpha,\beta) = (\mathbb{F}(\alpha))(\beta)$ and use induction (Homework).
\end{proof}

As a corollary of the above proposition, when $\alpha_1, \dots, \alpha_k \in \mathbb{L}$ are all algebraic over $\mathbb{F}$, the field extension $\mathbb{F}(\alpha_1, \dots, \alpha_k)$ can be constructed inductively as follows:
Define a sequence of fields by $\mathbb{F}_0 = \mathbb{F}$ and
\[ \mathbb{F}_{i+1} := \mathbb{F}_i(\alpha_{i+1}) \cong \frac{\mathbb{F}_i[x]}{\langle m^{\mathbb{F}_i}_{\alpha_{i+1}}(x) \rangle} \quad \text{for } 0 \le i \le k-1. \]
where $m^{\mathbb{F}_i}_{\alpha_{i+1}}(x) \in \mathbb{F}_i[x]$ is the minimal polynomial of the element $\alpha_{i+1}$ over the field $\mathbb{F}_i$. Then $\mathbb{F}_k = \mathbb{F}(\alpha_1, \dots, \alpha_k)$.

\section{Algebraic Extension}
We have seen that a simple extension $\mathbb{F}(\beta):\mathbb{F}$ contains an algebraic element $\beta$. Are the the other elements in $\mathbb{F}(\beta)$ also algebraic? More generally, how about the non-simple extension $\mathbb{F}(\beta_1, \dots, \beta_k)$?

\begin{definition}
An extension $\mathbb{E}:\mathbb{F}$ is called \textbf{algebraic} if every element $\alpha \in \mathbb{E}$ is algebraic over $\mathbb{F}$.
\end{definition}

\begin{theorem}
If $[\mathbb{E}:\mathbb{F}] < \infty$ (i.e., it is a finite extension), then $\mathbb{E}:\mathbb{F}$ is an algebraic extension.
\end{theorem}
\begin{proof}
Suppose $[\mathbb{E}:\mathbb{F}]=n$. Then for any $\alpha \in E$, consider the set $\{1, \alpha, \alpha^2, \dots, \alpha^n\}$. This is a set of $n+1$ elements in an $n$-dimensional vector space, so it must be linearly dependent over $\mathbb{F}$.
This implies there exist scalars $c_0, \dots, c_n \in \mathbb{F}$, not all zero, such that $c_0 \cdot 1 + c_1 \alpha + \dots + c_n \alpha^n = 0$.
Thus, $\alpha$ is a root of the non-zero polynomial $p(x) = c_nx^n + \dots + c_0 \in \mathbb{F}[x]$, which means $\alpha$ is algebraic over $\mathbb{F}$. Since $\alpha$ was arbitrary, the extension is algebraic.
\end{proof}

\begin{corollary}
Let $\mathbb{E}:\mathbb{F}$ be a field extension, and $\beta \in \mathbb{E}$ is algebraic over $\mathbb{F}$. Then the simple extension $\mathbb{F}(\beta):\mathbb{F}$ is algebraic.
\end{corollary}
\begin{proof}
By Kronecker's Theorem, $[\mathbb{F}(\beta):\mathbb{F}] = \deg(m_\beta(x))$ which is finite. So the result follows.
\end{proof}

One may ask how about the more general extension $\mathbb{F}(\beta_1, \dots, \beta_k):\mathbb{F}$ for algebraic $\beta_1, \dots, \beta_k \in \mathbb{E}$? We give a (partial) answer by the very powerful theorem below:
\begin{theorem}[Primitive Element Theorem]
Let $\mathbb{F}$ be a field with characteristic $\text{char}(\mathbb{F})=0$. Let $a, b \in \mathbb{K}$ be algebraic over $\mathbb{F}$ in some extension $\mathbb{K}:\mathbb{F}$. Then there exists an element $c \in \mathbb{F}(a,b)$ such that $\mathbb{F}(c) = \mathbb{F}(a,b)$. (Such an element $c$ is called a primitive element for the extension).
\end{theorem}

\begin{proof}
Let $p(x)$ be the minimal polynomial of $a$ over $\mathbb{F}$.
Let $q(x)$ be the minimal polynomial of $b$ over $\mathbb{F}$.

Take a field extension $\mathbb{L}:\mathbb{K}$ such that $p(x)$ and $q(x)$ split completely in $\mathbb{L}[x]$.
Suppose $a_1, \dots, a_m \in \mathbb{L}$ are the roots of $p(x)$, with $a_1=a$.
Suppose $b_1, \dots, b_n \in \mathbb{L}$ are the roots of $q(x)$, with $b_1=b$.

Since $\text{char}(\mathbb{F})=0$, the field $\mathbb{F}$ is infinite. This allows us to choose an element $d \in \mathbb{F}$ such that
\[ d \neq \frac{a_i - a_j}{b_k - b_l} \]
for all $i, j$ and for all $k \neq l$. In particular, we choose $d \in \mathbb{F}$ such that for any $i$ and any $j > 1$:
\[ d \neq \frac{a_i - a}{b - b_j} \]
(Note that the elements on the right-hand side may not be in $\mathbb{F}$, but there are only a finite number of such values, so we can always find a $d \in \mathbb{F}$ that avoids them).

Consider the element $c := a + db$.
Then, it is clear that $\mathbb{F}(c) \subseteq \mathbb{F}(a,b)$ since $a,b \in \mathbb{F}(a,b)$ and $d \in \mathbb{F} \subseteq \mathbb{F}(a,b)$.

To show the other inclusion, $\mathbb{F}(a,b) \subseteq \mathbb{F}(c)$, it is enough to show that $b \in \mathbb{F}(c)$, because if $b \in \mathbb{F}(c)$, then $a = c - db \in \mathbb{F}(c)$ as well.

Consider the polynomials $r(x) = p(c-dx)$ and $q(x)$, both viewed as polynomials in $\mathbb{F}(c)[x]$.
We evaluate both polynomials at $b$:
\begin{itemize}
    \item $r(b) = p(c-db) = p(a) = 0$.
    \item $q(b) = 0$.
\end{itemize}
Let $m(x) \in \mathbb{F}(c)[x]$ be the minimal polynomial of $b$ over the field $\mathbb{F}(c)$.
Since $b$ is a root of both $r(x)$ and $q(x)$ (which are in $\mathbb{F}(c)[x]$), it must be that $m(x)$ divides both $r(x)$ and $q(x)$ in $\mathbb{F}(c)[x]$.
\[ m(x) \mid r(x) \quad \text{and} \quad m(x) \mid q(x) \]
Now let's study the polynomial $m(x) \in \mathbb{F}(c)[x] \subseteq \mathbb{L}[x]$.

Since $m(x) \mid q(x)$ and $q(x)$ splits in $\mathbb{L}[x]$, the roots of $m(x)$ must be a subset of the roots of $q(x)$, which are $\{b = b_1, b_2, \dots, b_n\}$.

Consider any root $b_j$ of $m(x)$. Since $m(x)$ also divides $r(x)$, this $b_j$ must also be a root of $r(x)$.
So, for such a $b_j$, we must have $r(b_j) = p(c - db_j) = 0$.
This implies that $c - db_j$ must be one of the roots of $p(x)$, say $a_i$.
\[ c - db_j = a_i \]
Substituting $c=a+db$:
\[ a + db - db_j = a_i \]
\[ d(b - b_j) = a_i - a \]
If $j \neq 1$, then $b \neq b_j$. We can write:
\[ d = \frac{a_i - a}{b - b_j} \]
But this contradicts our initial choice of $d$, which was chosen specifically to not be equal to any of these values.
Therefore, the only possibility is that $j=1$, which means $b_j = b_1 = b$, and hence 
$$m(x) = x - b$$
in $\mathbb{F}(c)[x]$, i.e. $b \in \mathbb{F}(c)$ and the theorem is proved.
\end{proof}
\begin{corollary}
Let $\mathbb{E}:\mathbb{F}$ be a field extension with $char(\mathbb{F}) = 0$, and $\beta_1, \dots, \beta_k \in \mathbb{E}$ are algebraic over $\mathbb{F}$. Then the extension $\mathbb{F}(\beta_1, \dots, \beta_k):\mathbb{F}$ is algebraic.
\end{corollary}
\begin{proof}
By repeatedly applying Primitive Element Theorem, 
$$\mathbb{F}(\beta_1, \beta_2, \beta_3,\dots, \beta_k) = \mathbb{F}(\beta_1,\beta_2) (\beta_3,\dots, \beta_k) = \mathbb{F}(c_2)(\beta_3,\dots, \beta_k) = \mathbb{F}(c_3)(\dots, \beta_k) = \cdots = \mathbb{F}(c_k).$$
Since the simple extension $\mathbb{F}(c_k) : \mathbb{F}$ is algebraic, the result follows.
\end{proof}
One may still find the above proof unsatisfactory, since it assumes $char(\mathbb{F}) = 0$. In the next section, we will show that the same corollary holds for any $\mathbb{F}(\beta_1, \beta_2, \beta_3,\dots, \beta_k) ; \mathbb{F}$ without any assumptions (Corollary \ref{cor:extextfinite}).

\section{Tower Law}
\begin{theorem}[Tower Law]
Let $\mathbb{F} \subseteq \mathbb{K} \subseteq \mathbb{L}$ be field extensions.
(e.g., $\mathbb{Q} \subseteq \mathbb{Q}(\sqrt{2}) \subseteq \mathbb{Q}(\sqrt{2}, \sqrt{3})$).
Then:
\[ [\mathbb{L}:\mathbb{F}] = [\mathbb{L}:\mathbb{K}] [\mathbb{K}:\mathbb{F}] \]
\end{theorem}

\begin{proof}
Let $\{e_1, \dots, e_n\}$ be a basis of $\mathbb{K}$ over $\mathbb{F}$. This means $[\mathbb{K}:\mathbb{F}] = n$.
Let $\{f_1, \dots, f_m\}$ be a basis of $\mathbb{L}$ over $\mathbb{K}$. This means $[\mathbb{L}:\mathbb{K}] = m$.

\textbf{Claim:} The set $\mathcal{B} = \{e_i f_j \mid 1 \le i \le n, 1 \le j \le m\}$ is a basis of $\mathbb{L}$ over $\mathbb{F}$.
If this claim is true, then the dimension of $\mathbb{L}$ over $\mathbb{F}$ is $n \cdot m$, which proves the theorem. We need to show that $\mathcal{B}$ is a spanning set and is linearly independent over $\mathbb{F}$.

\textbf{Spanning set:}
Let $x$ be any element in $\mathbb{L}$. Since $\{f_j\}$ is a basis of $\mathbb{L}$ over $\mathbb{K}$, we can write $x$ as a linear combination:
\[ x = \sum_{j=1}^{m} \mu_j f_j, \quad \text{where each } \mu_j \in \mathbb{K} \]
Now, for each coefficient $\mu_j \in \mathbb{K}$, we can express it as a linear combination of the basis elements of $\mathbb{K}$ over $\mathbb{F}$. For each $j$, we have:
\[ \mu_j = \sum_{i=1}^{n} \nu_{ij} e_i, \quad \text{where each } \nu_{ij} \in \mathbb{F} \]
Substituting this back into the expression for $x$:
\[ x = \sum_{j=1}^{m} \left( \sum_{i=1}^{n} \nu_{ij} e_i \right) f_j = \sum_{j=1}^{m} \sum_{i=1}^{n} \nu_{ij} (e_i f_j) \]
Since every element $x \in \mathbb{L}$ can be written as a linear combination of the elements $\{e_i f_j\}$ with coefficients $\nu_{ij} \in \mathbb{F}$, this set spans $\mathbb{L}$ over $\mathbb{F}$.

\textbf{Linear Independence:}
Suppose we have a linear combination of the elements $\{e_i f_j\}$ that equals zero, with coefficients $\beta_{ij}$ from the field $\mathbb{F}$:
\[ \sum_{i=1}^{n} \sum_{j=1}^{m} \beta_{ij} e_i f_j = 0, \quad \text{where } \beta_{ij} \in \mathbb{F} \]
We can regroup the terms by factoring out the $f_j$:
\[ \sum_{j=1}^{m} \left( \sum_{i=1}^{n} \beta_{ij} e_i \right) f_j = 0 \]
For each $j$, the inner sum $\sum_{i=1}^{n} \beta_{ij} e_i$ is a linear combination of elements of the basis $\{e_i\}$ with coefficients in $\mathbb{F}$. This means the term in the parenthesis is an element of the field $\mathbb{K}$.
So, we have a linear combination of the basis elements $\{f_j\}$ with coefficients from $\mathbb{K}$ that equals zero.
By the linear independence of the set $\{f_1, \dots, f_m\}$ over $\mathbb{K}$, all the coefficients must be zero:
\[ \sum_{i=1}^{n} \beta_{ij} e_i = 0 \quad \text{for all } j = 1, \dots, m \]
Now, for each $j$, we have a linear combination of the basis elements $\{e_i\}$ over $\mathbb{F}$ that equals zero.
By the linear independence of the set $\{e_1, \dots, e_n\}$ over $\mathbb{F}$, all the coefficients must be zero:
\[ \beta_{ij} = 0 \quad \text{for all } i,j \]
This shows that the set $\{e_i f_j\}$ is linearly independent over $\mathbb{F}$, which completes the proof of the Tower Law. 
\end{proof}

\begin{example}
Let $\mathbb{F} = \mathbb{F}_2$ (which is the same as $\mathbb{Z}_2$).
Let $L = \mathbb{F}_2[x]/\langle x^3+x+1 \rangle$.
So, $|L| = 2^3 = 8$ and $[L:\mathbb{F}] = 3$.

Are there any proper subfields $\mathbb{K}$ with $\mathbb{F}_2 \subset \mathbb{K} \subset \mathbb{L}$?
Suppose such a subfield $\mathbb{K}$ exists. Then by the Tower Law, we must have:
\[ [\mathbb{L}:\mathbb{F}_2] = [\mathbb{L}:\mathbb{K}][\mathbb{K}:\mathbb{F}_2] \]
Substituting the known value:
\[ 3 = [\mathbb{L}:\mathbb{K}][\mathbb{K}:\mathbb{F}_2] \]
Since 3 is a prime number and the degrees of extensions are integers greater than or equal to 1, the only possibilities are:
\begin{enumerate}
    \item $[\mathbb{L}:\mathbb{K}]=1$ and $[\mathbb{K}:\mathbb{F}_2]=3$. This implies $\mathbb{K}=\mathbb{L}$.
    \item $[\mathbb{L}:\mathbb{K}]=3$ and $[\mathbb{K}:\mathbb{F}_2]=1$. This implies $\mathbb{K}=\mathbb{F}_2$.
\end{enumerate}
Therefore, there are no proper intermediate subfields between $\mathbb{L}$ and $\mathbb{F}_2$.
\end{example}




\begin{corollary} \label{cor:extextfinite}
Suppose we have a tower of fields $\mathbb{F} \subseteq \mathbb{E} \subseteq \mathbb{K}$. If $\mathbb{K}:\mathbb{E}$ is an algebraic extension and $\mathbb{E}:\mathbb{F}$ is an algebraic extension, then $\mathbb{K}:\mathbb{F}$ is also an algebraic extension.
\end{corollary}

\begin{proof}
Let $a \in \mathbb{K}$. Since $\mathbb{K}:\mathbb{E}$ is algebraic, $a$ is a root of some polynomial $p(x) = b_n x^n + \dots + b_1 x + b_0 \in \mathbb{E}[x]$.

Since $\mathbb{E}:\mathbb{F}$ is algebraic, each coefficient $b_i \in \mathbb{E}$ is algebraic over $\mathbb{F}$.
Consider the tower of fields:
\[ \mathbb{F} \subseteq \mathbb{F}_0 := \mathbb{F}(b_0) \subseteq \mathbb{F}_1 := \mathbb{F}_0(b_1) \subseteq \dots \subseteq \mathbb{F}_n := \mathbb{F}_{n-1}(b_n) = \mathbb{F}(b_0, \dots, b_n). \]
Since each $b_i$ is algebraic over $\mathbb{F}$, it is also algebraic over $\mathbb{F}_{i-1}$. Therefore, each extension in this tower, $\mathbb{F}_i : \mathbb{F}_{i-1}$, is a finite extension. By the Tower Law, $[\mathbb{F}_n : \mathbb{F}]$ is finite.
\[ [\mathbb{F}_n:\mathbb{F}] = [\mathbb{F}_n:\mathbb{F}_{n-1}] \dots [\mathbb{F}_1:\mathbb{F}_0][\mathbb{F}_0:\mathbb{F}] < \infty. \]
Now, the polynomial $p(x)$ has its coefficients in $\mathbb{F}_n$, so $p(x) \in \mathbb{F}_n[x]$. Since $a$ is a root of $p(x)$, the extension $\mathbb{F}_n(a):\mathbb{F}_n$ is finite, with $[\mathbb{F}_n(a):\mathbb{F}_n] \le \deg(p) = n$.

Consider the degree of the extension $\mathbb{F}_n(a)$ over $\mathbb{F}$:
\[ [\mathbb{F}_n(a):\mathbb{F}] = [\mathbb{F}_n(a):\mathbb{F}_n][\mathbb{F}_n:\mathbb{F}]. \]
This product is finite.
Since $\mathbb{F}(a)$ is a subfield of $\mathbb{F}_n(a)$, we have $[\mathbb{F}(a):\mathbb{F}] \le [\mathbb{F}_n(a):\mathbb{F}] < \infty$.
Because $\mathbb{F}(a):\mathbb{F}$ is a finite extension, it must be an algebraic extension. This implies that the element $a$ is algebraic over $\mathbb{F}$.
Since $a$ was an arbitrary element of $\mathbb{K}$, the extension $\mathbb{K}:\mathbb{F}$ is algebraic.
\end{proof}

\begin{corollary}
The set of all algebraic elements in an extension $\mathbb{E}:\mathbb{F}$ forms a subfield of $\mathbb{E}$. 
\end{corollary}
\begin{proof}
Let $a, b \in \mathbb{E}$ be algebraic over $\mathbb{F}$. We need to show that $a+b$, $a-b$, $ab$, and $a/b$ (for $b \neq 0$) are also algebraic over $\mathbb{F}$.
Consider the field $\mathbb{F}(a,b)$. By the Tower Law,
\[ [\mathbb{F}(a,b):\mathbb{F}] = [\mathbb{F}(a,b):\mathbb{F}(a)][\mathbb{F}(a):\mathbb{F}]. \]
Since $a$ is algebraic over $\mathbb{F}$, $[\mathbb{F}(a):\mathbb{F}]$ is finite.
Since $b$ is algebraic over $\mathbb{F}$, it is also algebraic over the larger field $\mathbb{F}(a)$. Thus, $[\mathbb{F}(a,b):\mathbb{F}(a)] = [\mathbb{F}(a)(b):\mathbb{F}(a)]$ is also finite.
Therefore, $[\mathbb{F}(a,b):\mathbb{F}]$ is finite.
Since $\mathbb{F}(a,b)$ is a finite extension of $\mathbb{F}$, it is an algebraic extension.
Any element of $\mathbb{F}(a,b)$ is algebraic over $\mathbb{F}$. This includes $a+b, a-b, ab,$ and $a/b$.
\end{proof}

To address these ancient Greek problems, we translate geometric constructions into the language of field theory.

\begin{definition}
A real number $\alpha$ is \textbf{constructible} if we can construct a line segment of length $|\alpha|$ using only a ruler and compass, starting from a segment of length 1.
\end{definition}

\begin{theorem}
If $\alpha \in \mathbb{R}$ is constructible, then the degree of the extension $[\mathbb{Q}(\alpha) : \mathbb{Q}]$ must be a power of $2$. 
\end{theorem}

\begin{proof}[Sketch]
Every point created by a ruler and compass is the intersection of two lines, a line and a circle, or two circles. Algebraically, these intersections involve solving linear or quadratic equations. Thus, every new point lies in a field $\mathbb{K}_{i+1}$ obtained by a quadratic extension of the previous field $\mathbb{K}_i$:
\[ \mathbb{Q} = K_0 \subset K_1 \subset K_2 \subset \dots \subset K_m \]
where $[K_{i+1} : K_i] = 2$ for all $i$. By the \textbf{Tower Law}:
\[ [K_m : \mathbb{Q}] = [K_m : K_{m-1}] \dots [K_1 : K_0] = 2^m \]
If $\alpha$ is constructed, then $\alpha \in \mathbb{K}_m$. Since $\mathbb{Q} \subseteq \mathbb{Q}(\alpha) \subseteq K_m$, the Tower Law again tells us that $[\mathbb{Q}(\alpha) : \mathbb{Q}]$ must divide $2^m$. Therefore, $[\mathbb{Q}(\alpha) : \mathbb{Q}] = 2^k$ for some $k \le m$.
\end{proof}


\subsection{Application 1: Trisecting the Angle}

We showed earlier that bisecting an angle is always possible. However, the Greeks could not find a way to divide an arbitrary angle into three equal parts. We show this is impossible by focusing on a specific angle: $60^\circ$.

\begin{proposition}
It is impossible to trisect an angle of $60^\circ$ using a ruler and compass.
\end{proposition}

\begin{proof}
Trisecting $60^\circ$ is equivalent to constructing the angle $20^\circ$. Constructing an angle $\theta$ is equivalent to constructing the number $\cos \theta$ on the $x$-axis. 
Recall the triple-angle formula for cosine:
\[ \cos(3\theta) = 4\cos^3 \theta - 3\cos \theta \]
Let $\theta = 20^\circ$, so $3\theta = 60^\circ$. Since $\cos 60^\circ = 1/2$, let $\alpha = \cos 20^\circ$. We have:
\[ \frac{1}{2} = 4\alpha^3 - 3\alpha \implies 8\alpha^3 - 6\alpha - 1 = 0 \]
To simplify, let $u = 2\alpha$. Substituting $\alpha = u/2$:
\[ 8\left(\frac{u}{2}\right)^3 - 6\left(\frac{u}{2}\right) - 1 = 0 \implies u^3 - 3u - 1 = 0 \]
Let $\mathbb{F}(u) = u^3 - 3u - 1$. By the Rational Root Theorem, the only possible rational roots are $\pm 1$. 
Checking them: $\mathbb{F}(1) = -3 \neq 0$ and $\mathbb{F}(-1) = 1 \neq 0$. 
Since $\mathbb{F}(u)$ is a cubic polynomial with no rational roots, it is irreducible over $\mathbb{Q}$.
Thus:
\[ [\mathbb{Q}(\alpha) : \mathbb{Q}] = [\mathbb{Q}(u) : \mathbb{Q}] = 3 \]
Because $3$ is not a power of $2$, $\cos 20^\circ$ is not constructible. Therefore, the angle $60^\circ$ cannot be trisected.
\end{proof}

This does not mean \textit{no} angle can be trisected (for example, a $90^\circ$ angle can be trisected because $30^\circ$ is constructible). It means there is no \textbf{general} method that works for any given angle.




\subsection{Application 2: Doubling the Cube}

The problem of doubling the cube (the Delian problem) asks if we can construct a cube with volume $2V$, given a cube of volume $V$. 

Suppose the original cube has side length $1$. Its volume is $1^3 = 1$. The side length $x$ of a cube with volume $2$ must satisfy $x^3 = 2$, so $x = \sqrt[3]{2}$.

\begin{proposition}
It is impossible to double the volume of a cube using a ruler and compass.
\end{proposition}

\begin{proof}
Let $x = \sqrt[3]{2}$. The polynomial $p(t) = t^3 - 2$ is irreducible over $\mathbb{Q}$ by Eisenstein's Criterion (with $p=2$). 
Since $p(t)$ is the minimal polynomial of $x$ over $\mathbb{Q}$, we have:
\[ [\mathbb{Q}(\sqrt[3]{2}) : \mathbb{Q}] = \deg(t^3 - 2) = 3 \]
Since $3$ is not a power of $2$, the number $\sqrt[3]{2}$ is not constructible. Thus, the side of the doubled cube cannot be constructed.
\end{proof}
